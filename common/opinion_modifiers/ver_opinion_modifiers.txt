# Serf's opinion of overlord
ver_triggered_opinion_serfdom = {
	trigger = {
		any_agreement = { agreement_preset = preset_ver_serf }
		FROM = {
			any_subject = { is_same_value = ROOT }
		}
	}

	opinion = {
		base = -100
	}
}

		}
